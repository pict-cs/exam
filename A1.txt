/**
 * Problem Statement: Write a non-recursive and recursive program to calculate Fibonacci
 * numbers and analyze their time and space complexity.
*/

#include <bits/stdc++.h>
using namespace std;

// Recursive Fibonacci
int fibonacci_recursive(int n){
    if(n <= 1) return n;
    return fibonacci_recursive(n - 1) + fibonacci_recursive(n - 2);
}

// Iterative Fibonacci
int fibonacci_iterative(int n){
    if(n <= 1) return n;

    int a = 0, b = 1, fib;
    for(int i = 2; i <= n; i++){
        fib = a + b;
        a = b;
        b = fib;
    }
    return fib;
}

int main(){
    int choice, n;

    while(true){
        cout << "\n===== Fibonacci Menu =====" << endl;
        cout << "1. Recursive Fibonacci" << endl;
        cout << "2. Iterative Fibonacci" << endl;
        cout << "3. Exit" << endl;
        cout << "Enter your choice: ";
        cin >> choice;

        switch(choice){
            case 1:
                cout << "Enter n: ";
                cin >> n;
                cout << "Fibonacci (Recursive) = " << fibonacci_recursive(n) << endl;
                break;

            case 2:
                cout << "Enter n: ";
                cin >> n;
                cout << "Fibonacci (Iterative) = " << fibonacci_iterative(n) << endl;
                break;

            case 3:
                cout << "Exiting program..." << endl;
                return 0;

            default:
                cout << "Invalid choice! Please try again." << endl;
        }
    }

    return 0;
}
